init.sql




CREATE TABLE IF NOT EXISTS domnitori (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nume VARCHAR(100),
    poza VARCHAR(255),
    an_inceput INT,
    an_sfarsit INT
);

CREATE TABLE IF NOT EXISTS log_modificari (
    id INT PRIMARY KEY AUTO_INCREMENT,
    domnitor_id INT,
    actiune VARCHAR(50),
    detalii TEXT,
    data_modificare TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

DELIMITER //

-- Proceduri Stocate
CREATE PROCEDURE sp_AdaugaDomnitor(IN p_nume VARCHAR(100), IN p_poza VARCHAR(255), IN p_start INT, IN p_final INT)
BEGIN
    INSERT INTO domnitori(nume, poza, an_inceput, an_sfarsit) VALUES (p_nume, p_poza, p_start, p_final);
END //

CREATE PROCEDURE sp_EditeazaDomnitor(IN p_id INT, IN p_nume VARCHAR(100), IN p_poza VARCHAR(255), IN p_start INT, IN p_final INT)
BEGIN
    UPDATE domnitori SET nume = p_nume, poza = p_poza, an_inceput = p_start, an_sfarsit = p_final WHERE id = p_id;
END //

CREATE PROCEDURE sp_StergeDomnitor(IN p_id INT)
BEGIN
    DELETE FROM domnitori WHERE id = p_id;
END //

-- Triggere
CREATE TRIGGER tr_DupaInsert AFTER INSERT ON domnitori
FOR EACH ROW BEGIN
    INSERT INTO log_modificari(domnitor_id, actiune) VALUES (NEW.id, 'ADAUGARE');
END //

CREATE TRIGGER tr_InainteUpdate BEFORE UPDATE ON domnitori
FOR EACH ROW BEGIN
    IF NEW.an_sfarsit < NEW.an_inceput THEN
        SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT = 'Eroare: Anul de sfarsit e mai mic decat inceputul!';
    END IF;
END //

CREATE TRIGGER tr_DupaDelete AFTER DELETE ON domnitori
FOR EACH ROW BEGIN
    INSERT INTO log_modificari(domnitor_id, actiune) VALUES (OLD.id, 'STERGERE');
END //

DELIMITER ;
